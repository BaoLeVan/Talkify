
# 07 — PostgreSQL
````
table users {
  id integer [primary key]
  avatar varchar
  username varchar
  full_name varchar
  email varchar
  role varchar
  password varchar
  status boolean
  isActive boolean
  verifyToken varchar
  deleted_at timestamp
  created_at timestamp
  updated_at timestamp
}

table device {
  id integer [primary key]
  device_name varchar
  mac_adress varchar
  device_token varchar
  user_id integer
}

table conversations {
  id integer [primary key]
  title varchar
  avatar_url varchar
  type enum('PRIVATE','GROUP')
  last_message_id varchar
  last_message_content varchar
  last_message_at varchar
  sequence_counter bigint
  deleted_at timestamp
  created_at timestamp
  updated_at timestamp
}

table conversation_members {
  users_id integer
  conversations_id integer
  role varchar
  nickname varchar
  last_read_sequence bigint
  updated_at timestamp
}
````
